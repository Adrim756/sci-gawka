7. Pętle

for

for($i = 0; $i < 5; $i++){
 echo $i;
}

while

while($x < 5){
 $x++;
}

foreach — najważniejsze dla tablic

$tab = [1,2,3];

foreach($tab as $el){
 echo $el;
}

17. MySQL

$conn = mysqli_connect(
 "localhost",
 "root",
 "",
 "test"
);

mysqli_query() — wykonuje zapytanie

mysqli_fetch_array() — pobiera rekord jako tablicę $row['imie'] bądź $row[0]

mysqli_fetch_assoc() — pobiera rekord jako tablica asocjacyjna $row['imie']

mysqli_fetch_row() — pobiera rekord jako zwykła tablica $row[0]

mysqli_num_rows() — liczba rekordów

mysqli_close() — zamyka połączenie
