1. Zmienne

var – stara zmienna, ma zasięg funkcji, hoisting (może powodować błędy)

let – nowoczesna zmienna, działa tylko w bloku { }

const – stała (nie można zmienić wartości)

2. Typy danych

string – tekst

number – liczby

boolean – true / false

null – brak wartości (celowo)

undefined – brak wartości (automatycznie)

object – obiekt

array – tablica

3. Operatory

=== – porównanie wartości i typu (NAJLEPSZE)

== – porównanie tylko wartości (unikać)

&& – AND

|| – OR

! – NOT

+ – dodawanie

- – odejmowanie

* – mnożenie

/ – dzielenie

% – dzielenie z resztą (modulo)

+= – dodaj i przypisz

-= – odejmij i przypisz

*= – pomnóż i przypisz

/= – podziel i przypisz

i++ – najpierw używa wartości, potem zwiększa o 1 (postinkrementacja)

++i – najpierw zwiększa o 1, potem używa wartości (preinkrementacja)

4. Stringi

.length – długość tekstu

.toUpperCase() – wielkie litery

.toLowerCase() – małe litery

.trim() – usuwa spacje

.includes() – sprawdza czy zawiera tekst

.split() – dzieli na tablicę

.join() – łączy tablicę

5. Tablice

.pop() – usuwa z końca

.shift() – usuwa z początku

.unshift() – dodaje na początek

.map() – tworzy nową tablicę

.filter() – filtruje elementy

.find() – znajduje pierwszy element

.forEach() – wykonuje dla każdego

async – funkcja asynchroniczna

Math.random() – losowa liczba

20. Debugowanie

console.log() — wypisuje dane

console.table() — tabela

debugger — zatrzymuje kod

Na egzaminie używaj: let i const

21. Warunki (if / else)

if (warunek) { } — wykonuje kod jeśli warunek jest prawdziwy (true)

if (x > 5) { console.log("większe"); } — przykład

if (warunek) { } else { } — jeśli warunek fałszywy wykona else

if (x > 5) { }
else if (x === 5) { }
else { } — wiele warunków

switch(x) — alternatywa dla wielu if

22. PSEUDO-KLASY STRUKTURALNE CSS

:first-child – pierwszy element w rodzicu

:last-child – ostatni element w rodzicu

:nth-child(n) – wybiera n-ty element (np. 2, 3, 2n)

:nth-child(2n) – co drugi element

:nth-child(odd) – elementy nieparzyste

:nth-child(even) – elementy parzyste

:first-of-type – pierwszy element danego typu

:last-of-type – ostatni element danego typu

:nth-of-type(n) – n-ty element danego typu

:only-child – element, który jest jedynym dzieckiem

:only-of-type – jedyny element danego typu

:not(selector) – wybiera wszystko oprócz wskazanego

23. Pętle (loops)

for — pętla z licznikiem

for(let i = 0; i < 5; i++){
 console.log(i);
}

while — wykonuje dopóki warunek true

let i = 0;
while(i < 5){
 i++;
}

do while — wykona się przynajmniej raz

let i = 0;
do{
 i++;
}while(i < 5);

for...of — do tablic (najczęściej na egzaminie)

const tab = [1,2,3];
for(const el of tab){
 console.log(el);
}

for...in — do obiektów

const obj = {a:1,b:2};
for(const key in obj){
 console.log(key //-klucz objektu//);
}

break — przerywa działanie pętli lub instrukcji switch

forEach działa tylko na tablicach!

for...in = obiekty

for...of = tablice (najlepsze)
